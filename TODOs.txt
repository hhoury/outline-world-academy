var axios = require('axios');
var FormData = require('form-data');
var data = new FormData();
data.append('email_field', 'ahmadhaidarahmad555@hotmail.com');
data.append('name_field', 'ahmadhaidarahmad24');

var config = {
  method: 'post',
  url: 'https://api.outlineworldacademy.com/api/newsletter/outlineacademy/subscribe/',
  headers: { 
    ...data.getHeaders()
  },
  data : data
};

axios(config)
.then(function (response) {
  console.log(JSON.stringify(response.data));
})
.catch(function (error) {
  console.log(error);
});
